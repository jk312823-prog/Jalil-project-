<script>
let expenses=JSON.parse(localStorage.getItem('expenses'))||[];
let balance=parseFloat(localStorage.getItem('balance'))||0;
const expenseForm=document.getElementById('expenseForm');
const expenseTable=document.getElementById('expenseTable');
const balanceDisplay=document.getElementById('balance');
balanceDisplay.textContent=balance;
function renderTable(){
expenseTable.innerHTML=`<tr><th>تاریخ</th><th>رقم</th><th>ذرائع</th><th>تفصیل</th></tr>`;
expenses.forEach(exp=>{
const row=expenseTable.insertRow();
row.insertCell(0).textContent=exp.date;
row.insertCell(1).textContent=exp.amount;
row.insertCell(2).textContent=exp.source;
row.insertCell(3).textContent=exp.notes;
});
}
renderTable();
expenseForm.addEventListener('submit',function(e){
e.preventDefault();
const amount=parseFloat(document.getElementById('amount').value);
const source=document.getElementById('source').value;
const notes=document.getElementById('notes').value;
const date=new Date().toLocaleDateString();
balance-=amount;
balanceDisplay.textContent=balance;
const expense={date,amount,source,notes};
expenses.push(expense);
localStorage.setItem('expenses',JSON.stringify(expenses));
localStorage.setItem('balance',balance);
renderTable();
expenseForm.reset();
});
</script>
